<script>
const products = [
  {id:1,title:'Wireless Earbuds',price:2499,country:'PK',vendorId:101},
  {id:2,title:'Smart Watch',price:4999,country:'IN',vendorId:102},
  {id:3,title:'Handmade Rug',price:7999,country:'TR',vendorId:103},
  {id:4,title:'Organic Tea Pack',price:899,country:'CN',vendorId:104},
  {id:5,title:'Leather Wallet',price:1299,country:'PK',vendorId:101},
  {id:6,title:'Designer Lamp',price:5999,country:'EG',vendorId:105}
];
const vendors = [
  {id:101,name:'Abubakar Traders',country:'Pakistan'},
  {id:102,name:'Indo Gadgets',country:'India'},
  {id:103,name:'Turk Rugs Co',country:'Turkey'},
  {id:104,name:'China Organic',country:'China'},
  {id:105,name:'Cairo Decor',country:'Egypt'}
];
let cart = JSON.parse(localStorage.getItem('prime_cart')||'[]');

function formatPrice(n){return 'Rs. '+n}
function getVendor(id){return vendors.find(v=>v.id===id)||{name:'Unknown'}}
function renderProducts(){
  const q = document.getElementById('searchInput').value.trim().toLowerCase();
  const grid = document.getElementById('productsGrid'); grid.innerHTML='';
  const filtered = products.filter(p=> p.title.toLowerCase().includes(q) || (getVendor(p.vendorId).name.toLowerCase().includes(q)));
  (filtered.length?filtered:products).forEach(p=>{
    const div = document.createElement('div'); div.className='card';
    div.innerHTML = `
      <img src="data:image/svg+xml;utf8,<svg xmlns='http://www.w3.org/2000/svg' width='600' height='300'><rect width='100%' height='100%' fill='%23eef5ff'/><text x='50%' y='50%' font-size='18' fill='%230b2540' text-anchor='middle' dominant-baseline='middle'>${encodeURIComponent(p.title)}</text></svg>" />
      <h3>${p.title}</h3>
      <div class="smalltext">Vendor: ${getVendor(p.vendorId).name} • ${p.country}</div>
      <div class="price">${formatPrice(p.price)}</div>
      <div class="actions">
        <button class="btn" onclick="addToCart(${p.id})">Add to Cart</button>
        <button class="btn secondary" onclick="viewProduct(${p.id})">View</button>
      </div>
    `;
    grid.appendChild(div);
  });
}
function renderVendors(){
  const g = document.getElementById('vendorsGrid'); g.innerHTML='';
  vendors.forEach(v=>{
    const d = document.createElement('div'); d.className='card';
    d.innerHTML = `<h3>${v.name}</h3><div class='smalltext'>${v.country}</div><div style='margin-top:8px'><button class='btn' onclick="viewVendor(${v.id})">View Shop</button></div>`;
    g.appendChild(d);
  });
}
function addToCart(id){
  const p = products.find(x=>x.id===id); if(!p) return;
  const existing = cart.find(i=>i.id===id);
  if(existing) existing.qty++;
  else cart.push({id:p.id,title:p.title,price:p.price,qty:1});
  saveCart(); renderCart(); alert('Added to cart: '+p.title);
}
function saveCart(){localStorage.setItem('prime_cart',JSON.stringify(cart))}
function renderCart(){
  const box = document.getElementById('cartItems'); box.innerHTML='';
  if(cart.length===0) box.innerHTML='<div class="smalltext">Cart is empty</div>';
  let total=0;
  cart.forEach(item=>{
    total += item.price*item.qty;
    const el = document.createElement('div'); el.className='cart-item';
    el.style.display='flex'; el.style.gap='8px'; el.style.alignItems='center'; el.style.padding='8px 0';
    el.innerHTML = `<div style='flex:1'><strong>${item.title}</strong><div class='smalltext'>${formatPrice(item.price)} x ${item.qty}</div></div><div><button class='btn secondary' onclick='changeQty(${item.id},-1)'>-</button> <button class='btn' onclick='changeQty(${item.id},1)'>+</button></div>`;
    box.appendChild(el);
  });
  document.getElementById('cartTotal').innerText = formatPrice(total);
}
function changeQty(id,delta){ const it = cart.find(i=>i.id===id); if(!it) return; it.qty += delta; if(it.qty<=0) cart = cart.filter(i=>i.id!==id); saveCart(); renderCart(); }
function checkout(){ if(cart.length===0){alert('Cart is empty'); return} alert('Demo checkout — This is a static demo. For real payments you need backend & payment gateway.'); cart = []; saveCart(); renderCart(); }
function viewProduct(id){ const p = products.find(x=>x.id===id); if(!p) return; alert(`${p.title}\\nPrice: ${formatPrice(p.price)}\\nVendor: ${getVendor(p.vendorId).name}`); }
function viewVendor(id){ const v = vendors.find(x=>x.id===id); if(!v) return; alert(`Vendor: ${v.name}\\nCountry: ${v.country}\\nThis is a demo vendor page.`) }
function showSection(id){ ['home','vendors','sell'].forEach(s=> document.getElementById(s).style.display = (s===id)?'block':'none'); if(id==='vendors') renderVendors(); }
renderProducts(); renderCart();
</script>
